
xlsx的type类型
base64: 以base64方式读取；


binary: BinaryString格式(byte n is data.charCodeAt(n))


string: UTF8编码的字符串；


buffer: nodejs Buffer；


array: Uint8Array，8位无符号数组；


file: 文件的路径（仅nodejs下支持）
xlsx读取workbook方法
XLSX.utils.sheet_to_csv：生成CSV格式


XLSX.utils.sheet_to_txt：生成纯文本格式


XLSX.utils.sheet_to_html：生成HTML格式


XLSX.utils.sheet_to_json：输出JSON格式

importExcel(e) {//导入
	let obj=e.target  //获取input对象
	var f = obj.files[0];//获取上传的文件
		var reader = new FileReader(); //创建文件读取实例
		reader.readAsBinaryString(f);//文件读取为二进制字符串
        reader.onload = function(e) {//读取成功后执行
            var data = e.target.result;//获取读取后结果
                wb = XLSX.read(data, {
                    type: 'binary'
				});
            		//wb.SheetNames[0]是获取Sheets中第一个Sheet的名字
			//wb.Sheets[Sheet名]获取第一个Sheet的数据
			let	aa=JSON.stringify( XLSX.utils.sheet_to_json(wb.Sheets[wb.SheetNames[0]]) );
			console.log(JSON.parse(aa))//转化为json格式
          

};  }

// FileReader共有4种读取方法：
        // 1.readAsArrayBuffer(file)：将文件读取为ArrayBuffer。
        // 2.readAsBinaryString(file)：将文件读取为二进制字符串
        // 3.readAsDataURL(file)：将文件读取为Data URL
        // 4.readAsText(file, [encoding])：将文件读取为文本，encoding缺省值为'UTF-8'
